types:
  # Common Response Types
  StandardResponse:
    type: object
    properties:
      code:
        type: string
        description: Response code (0 indicates success)
        example: "0"
      msg:
        type: string
        description: Response message
        example: "success"
      data:
        type: any
        description: Response payload

  ErrorResponse:
    type: object
    properties:
      code:
        type: string
        description: Error code
      msg:
        type: string
        description: Error message

  # Time Series Data Types
  OHLCData:
    type: object
    properties:
      time:
        type: integer
        format: int64
        description: Timestamp in milliseconds
      open:
        type: number
        description: Opening value
      high:
        type: number
        description: High value
      low:
        type: number
        description: Low value
      close:
        type: number
        description: Closing value

  TimeValueData:
    type: object
    properties:
      time:
        type: integer
        format: int64
        description: Timestamp in milliseconds
      value:
        type: number
        description: Data value

  # Market Data Types
  MarketData:
    type: object
    properties:
      symbol:
        type: string
      price:
        type: number
      price_change:
        type: number
      price_change_percent:
        type: number
      volume:
        type: number
      open_interest:
        type: number
      funding_rate:
        type: number

  # Common Query Parameters
  CommonTimeParams:
    type: object
    properties:
      startTime:
        type: integer
        format: int64
        required: false
        description: Start timestamp in milliseconds
      endTime:
        type: integer
        format: int64
        required: false
        description: End timestamp in milliseconds
      limit:
        type: integer
        required: false
        description: Number of results (max varies by endpoint)
        default: 100
